//Операторы:
//1. Проверка на корректность while и return 
//(после оператора при необходдимости, должен измениться тип возвращаемого значения
//и jump на epilog на случай, если return вызван не в конце программы)
int while_test(){
	float a = 12;
	while(a){
		while_test();
		return a;
	}
	int j = a * 1e1 + 12;
	return j;
}
	
int main(){
	return while_test();
}
